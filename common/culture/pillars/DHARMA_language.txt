
language_sanskrit = {# Sanskrit
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_sanskrit
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_sanskrit }
			multiply = 10
		}
	}

	color = kashmiri
}
