#Communion & Denominations
declare_autocephaly_scripted_effect = {
	root.faith = {
		change_fervor = {
			value = 20
			desc = fervor_gain_autocephaly
		}
		add_doctrine = orthodox_autocephaly_autocephalous
	}

	faith:orthodox = {
		change_fervor = {
			value = -15
			desc = fervor_loss_autocephaly
		}
	}
}

reconcile_with_rome_scripted_effect = {
	root.faith = {
		change_fervor = {
			value = 5
			desc = fervor_gain_reconciled_with_rome
		}
		remove_doctrine = special_doctrine_orthodox_communion
		add_doctrine = special_doctrine_catholic_communion
	}

	faith:catholic = {
		change_fervor = {
			value = 10
			desc = fervor_gain_reconciled_with_rome
		}
	}
}

restore_roman_patriarchate_scripted_effect = {
	faith:roman_orthodox = {
		add_doctrine = orthodox_autocephaly_pentarch
		add_doctrine = special_doctrine_ecumenical_christian
	}
	religion:christianity_religion = {
		every_faith = {
			limit = {
				this.religion = root.faith.religion
				has_doctrine = special_doctrine_orthodox_communion
			}
			change_fervor = {
				value = 10
				desc = fervor_loss_roman_patriarchate
			}
		}
	}

	every_ruler = {
		limit = {
			faith = faith:catholic
		}
		trigger_event = autocephaly_decision.0021
	}
	faith:catholic = {
		change_fervor = {
			value = -25
			desc = fervor_loss_roman_patriarchate
		}
	}
}

establish_latin_jersualem_scripted_effect = {
	faith:catholic = {
		change_fervor = {
			value = 10
			desc = fervor_gain_jerusalem_patriarchate
		}
	}
	faith:jerusalem_melkite = {
		change_fervor = {
			value = -15
			desc = fervor_loss_jerusalem_patriarchate
		}
		remove_doctrine = special_doctrine_ecumenical_christian
	}
	title:d_jerusalem_melkite_patriarchate = {
		if = {
			limit = {
				title:e_byzantium = { exists = yes }
			}
			change_liege = {
				liege = e_byzantium
				change = d_jerusalem_melkite_patriarchate
			}
		}
		else = {
			destroy_title = d_jerusalem_melkite_patriarchate
		}
	}
}

restore_school_of_edessa_scripted_effect = {
	religion:christianity_religion = {
		every_faith = {
			limit = {
				AND = {
					has_doctrine = doctrine_christology_nestorian
					NOT = { faith = faith:messalian }
				}
			}
			remove_doctrine = special_doctrine_independent_communion
			remove_doctrine = special_doctrine_oriental_communion
			add_doctrine = special_doctrine_church_of_east
			activate_holy_site = edessa
		}
	}
}
