war_prop_lower = {
	value = 0
	if = {
		limit = { house.var:titular_republic.var:republic_owner.max_military_strength > scope:recipient.double_military_strength }
		add = house.var:titular_republic.var:republic_owner.max_military_strength
		subtract = scope:recipient.double_military_strength
	}
	else_if = {
		limit = { house.var:titular_republic.var:republic_owner.double_military_strength < scope:recipient.max_military_strength }
		add = house.var:titular_republic.var:republic_owner.max_military_strength
		subtract = scope:recipient.max_military_strength
	}
	else = {
		add = house.var:titular_republic.var:republic_owner.max_military_strength
		subtract = scope:recipient.max_military_strength
		divide = 2
	}
	if = { limit = { house.var:titular_republic = { has_variable = very_warlike } } add = 175 }
	if = { limit = { house.var:titular_republic = { has_variable = warlike } } add = 70 }
	if = { limit = { house.var:titular_republic = { has_variable = peaceful } } subtract = 70 }
	if = { limit = { house.var:titular_republic = { has_variable = very_peaceful } } subtract = 500 }
}
