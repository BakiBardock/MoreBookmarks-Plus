cow_vineyards = {
	value =  0
	every_directly_owned_province = {
		limit = {
			AND = {
				has_holding_type = castle_holding
				OR = {
					building_slots <= 7
					AND = {
						has_free_building_slot = yes
						building_slots > 7
					}
				}
				NOT = { has_building_or_higher = vineyard_01 }
				OR = {
					terrain = drylands
					terrain = hills
					terrain = farmlands
					terrain = plains
					terrain = floodplains
				}
			}
		}
		add = 1
	}
	every_directly_owned_province = {
		limit = {
			AND = {
				has_holding_type = city_holding
				OR = {
					building_slots <= 7
					AND = {
						has_free_building_slot = yes
						building_slots > 7
					}
				}
				NOT = { has_building_or_higher = vineyard_01 }
				OR = {
					terrain = drylands
					terrain = hills
					terrain = farmlands
					terrain = plains
					terrain = floodplains
				}
			}
		}
		add = 1
	}
	every_directly_owned_province = {
		limit = {
			has_holding_type = church_holding
			barony = {
				NOT = { exists = lessee }
			}
			OR = {
				building_slots <= 7
				AND = {
					has_free_building_slot = yes
					building_slots > 7
				}
				}
			NOT = { has_building_or_higher = vineyard_01 }
				OR = {
					terrain = drylands
					terrain = hills
					terrain = farmlands
					terrain = plains
					terrain = floodplains
				}
		}
		add = 1
	}
